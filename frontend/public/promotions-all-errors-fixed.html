🛠️ Comprehensive Fixes Applied

Fix #1: Value Display Function

// BEFORE ❌
const getValueDisplay = (promotion: Promotion) => {
  switch (promotion.type) {
    default:
      return promotion.value.toString(); // CRASH!
  }
};

// AFTER ✅
const getValueDisplay = (promotion: Promotion) => {
  // Handle undefined value
  if (promotion.value === undefined || promotion.value === null) {
    return 'Not specified';
  }
  
  switch (promotion.type) {
    default:
      return promotion.value.toString(); // Safe!
  }
};

Fix #2: Usage Count Display

// BEFORE ❌
{promotion._count.usage}

// AFTER ✅
{promotion._count?.usage || promotion.usageCount || 0}

Fix #3: TypeScript Interface Updates

// BEFORE ❌
interface Promotion {
  value: number; // Required but API returns undefined
  _count: {
    usage: number; // Required but API returns undefined
  };
}

// AFTER ✅
interface Promotion {
  value?: number; // Optional
  _count?: {      // Optional
    usage: number;
  };
}
